Function with Context and Dependencies

This example demonstrate how to pass both optional arguments: context and dependency. 

function Foobar(arr1,arr2){
	this.arr1 = arr1;
	this.arr2 = arr2;
}
Foobar.prototype.union = function(){
	return _.union(this.arr1,this.arr2);
}

var myFoobar = new Foobar([1,2,3],[2,3,4]);



var baz = myFoobar.union();
console.log(baz);


/* Callback style */
vkthread.exec(myFoobar.union, //function to run in a thread;
		[], //arguments for the function;
		function(data){ // callback function;
			var baz = data;
			console.log(baz);
		},
		myFoobar, // context
		['../js/underscore-min.js'] /* dependency */
	);

/* Promise style */	
vkthread.run(myFoobar.union, //function to run in a thread;
		[], //arguments for the function;
		myFoobar, // context
		['../js/underscore-min.js'] /* dependency */
	).then(function(data){
		var baz = data;
		console.log(baz);
	});
