function dummySum(start,end){
	var sum = 0,
	    ix,
	    dummy = 'a,b,c';
		
	for(var ix=start; ix<end; ix++){
		dummy.split(',').join('-');
		sum = sum+ix;
	}
	return sum;
} 
function runAll_dummySum(){

	vkthread.runAll([ [dummySum, [0,  3e6]], 
			  [dummySum, [3e6,7e6]] ]
			).then(function (data) {
		document.getElementById('demo_result')
				.innerHTML = data[0]+data[1];
		})
}
