Standalone Function
function foo(str) {
	var ret, ix;
	for(ix=0;ix<7e6;ix++){
		ret = str.split(',').join('-');
	}
	return ret;
}
function run_foo(){
	document.getElementById('demo_result').innerHTML = foo('a,b,c');
}

function run_foo_in_thread(){
	vkthread.exec(foo,['x,y,z'], function(data){
		document.getElementById('demo_result_thread').innerHTML = data;
	});
}
