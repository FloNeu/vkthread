Anonymous Function

Anonymous function also can be run in a thread, why not?




var param = {
	fn: function(ar1, ar2){
		return _.union(ar1,ar2)
	},
	args: [[1,2,3], [3,4,5]],
	importFiles: ['https://cdnjs.cloudflare.com/ajax/libs/underscore.js/1.8.3/underscore-min.js']
}

vkthread.exec(param).then(
	function(data){
		console.log(data);
	}
);
