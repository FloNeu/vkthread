Anonymous Function

Anonymous function also can be run in a thread, why not?

var foo = (function(ar1,ar2){
		return _.union(ar1,ar2);
	})([1,2,3],[3,4,5]);
	
console.log(foo);

/* Callback style */
vkthread.exec(function(ar1,ar2){return _.union(ar1,ar2)}, //anonymous function
		[[1,2,3],[3,4,5]], //list of arguments
		function(data){ // callback function
			var foo = data;
			console.log(foo);
		}, 
		['../js/underscore-min.js'] 
	);

/* Promise style */	
vkthread.run(function(ar1,ar2){return _.union(ar1,ar2)}, //anonymous function
		[[1,2,3],[3,4,5]], //list of arguments
		['../js/underscore-min.js'] 
	).then(function(data){ 
		var foo = data;
		console.log(foo);
	});
