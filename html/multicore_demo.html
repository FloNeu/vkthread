Multi-Core Processing

This page demonstrates an advantage of multi-threading on multi-core CPU. The first example runs dummySum() function in a single thread. The second one divides this function in 2 tasks and runs a separate thread for each of them. At OS kernel level these threads are executed in 2 CPU cores simultaneously, which makes this application almost 2 times faster.

However, in javascript we don't have access to OS, so we can't guarantee, that browser forces OS to run threads on different cores; both threads can be run on the same core. In this case we won't see significant advantage. You can run this page in different browsers to compare performance and multicore feature. 

function dummySum(start,end){
	var sum =0, ix, dummy = 'a,b,c';
		
	for(ix=start; ix<end; ix++){
		dummy.split(',').join('-');
		sum = sum+ix;
	}
	return sum;
} 
function run_dummySum_time(){

var d1= +(new Date());
	
vkthread.run(dummySum, [0,8e6])
	.then(function (data) {
		var timeSpan = (+(new Date())-d1)/1000;
		document.getElementById('demo_result')
			.innerHTML = data + ' : ' + timeSpan + ' sec';
	})
}

function runAll_dummySum_time(){

var d1= +(new Date());
	
vkthread.runAll([ 
		  [dummySum, [0, 4e6]], 
		  [dummySum, [4e6, 8e6]] 
		]).then(function (data) {
			var timeSpan = (+(new Date())-d1)/1000;
			document.getElementById('demo_result_thread')
				.innerHTML = data[0]+data[1] + ' : ' + timeSpan + ' sec';
		})
	}
