Lambda Function

Lambda. Have fun! 

function greetLambda(param) {
	var displayMessage = (function(msg1){  
		return function(msg2){  
			return msg1 + msg2;
	   }     
	}(param));
	return displayMessage("Lambda World!");
}
var foo = greetLambda('Hello, ');
console.log(foo);

/* Callback style */
vkthread.exec(greetLambda, // lambda function to run in a thread;
		['Hello, '], //arguments for the function;
		function(data){ // callback
			var foo = data;
			console.log(foo);
		}
	);

/* Promise style */		
vkthread.run(greetLambda, // lambda function to run in a thread;
		['Hello, '] //arguments for the function;
		).then(function(data){ 
			var foo = data;
			console.log(foo);
		});
