function greetLambda(arg) {
    var displayMessage = (function(msg1){
        return (msg2) => msg1 + msg2; // ES6 Arrow function
    }(arg));
    return displayMessage("Lambda World!");
}




var param = {
    fn: greetLambda,
    args: ['Hello, ']
}
