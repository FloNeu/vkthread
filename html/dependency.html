Function with Dependencies

If your function depends on library(ies), thread needs to know path to the library(ies). 

//function "_.union()" belongs to "Underscore" library.
function bar(arr1,arr2){
	return _.union(arr1,arr2);
}
var foo = bar([1,2,3],[2,3,4]);

console.log(foo);

/* Callback style */
vkthread.exec(bar, //function to run in a thread;
		[[1,2,3],[2,3,4]], //arguments for the function;
		function(data){ // callback function;
			var foo = data;
			console.log(foo);
		},
		['../js/underscore-min.js'] /* dependency */
	);

/* Promise style */
vkthread.run(bar, //function to run in a thread;
	    [[1,2,3],[2,3,4]], //arguments for the function;
	    ['../js/underscore-min.js'] /* dependency */
	).then(function(data){
		var foo = data;
		console.log(foo);
	});
